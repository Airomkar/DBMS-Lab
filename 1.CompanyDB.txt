SOLUTION-- 
            
Step 1: Create Database

CREATE DATABASE company;
USE company;

Step 2: Create Tables

CREATE TABLE dept (
  deptno INT PRIMARY KEY,
  deptname VARCHAR(30),
  location VARCHAR(30)
);

CREATE TABLE emp (
  eno INT PRIMARY KEY,
  ename VARCHAR(30),
  job VARCHAR(20),
  hiredate DATE,
  salary DECIMAL(10,2),
  commission DECIMAL(10,2),
  deptno INT,
  FOREIGN KEY (deptno) REFERENCES dept(deptno)
);

1. List the maximum salary paid to salesman  

SELECT MAX(salary) AS Max_Salary_Salesman
FROM emp
WHERE job = 'Salesman';

2. List name of emp whose name start with â€˜Iâ€™.

SELECT ename
FROM emp
WHERE ename LIKE 'I%';

3. List details of emp who have joined before â€™30-sept-81â€™

SELECT *
FROM emp
WHERE hiredate < '1981-09-30';

4. List the emp details in the descending order of their basic salary 

SELECT *
FROM emp
ORDER BY salary DESC;

5. List of no. of emp & avg salary for emp in the dept no â€˜20â€™

SELECT COUNT(*) AS No_of_Employees,
       AVG(salary) AS Avg_Salary
FROM emp
WHERE deptno = 20;

6. List the avg salary, minimum salary of the emp hiredatewise for dept no â€˜10â€™. 

SELECT hiredate,
       AVG(salary) AS Avg_Salary,
       MIN(salary) AS Min_Salary
FROM emp
WHERE deptno = 10
GROUP BY hiredate;

7. List emp name and its department

SELECT e.ename, d.deptname
FROM emp e
JOIN dept d ON e.deptno = d.deptno;

8. List total salary paid to each department 

SELECT d.deptname, SUM(e.salary) AS Total_Salary
FROM emp e
JOIN dept d ON e.deptno = d.deptno
GROUP BY d.deptname;

9. List details of employee working in â€˜Devâ€™ department 

SELECT e.*
FROM emp e
JOIN dept d ON e.deptno = d.deptno
WHERE d.deptname = 'Dev';

10. Update salary of all employees in deptno 10 by 5 %. 

UPDATE emp
SET salary = salary * 1.05
WHERE deptno = 10;
