3️⃣ Find information about two teachers

db.Teachers.find().limit(2).pretty()

4️⃣ Find all teachers of computer department

db.Teachers.find({ dname: "COMP" }).pretty()

5️⃣ Find all teachers of Computer, IT, and E&TC departments

db.Teachers.find({ dname: { $in: ["COMP", "IT", "E&TC"] } }).pretty()

6️⃣ Find teachers of Computer, IT, and E&TC departments having salary ≥ 25000

db.Teachers.find({
  dname: { $in: ["COMP", "IT", "E&TC"] },
  salary: { $gte: 25000 }
}).pretty()
